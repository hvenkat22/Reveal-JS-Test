import io, base64, textwrap, os
import pandas as pd
import matplotlib.pyplot as plt
import seaborn as sns

sns.set_style("whitegrid")
sns.set_context("talk")

csv_data = """employee_id,department,region,performance_score,years_experience,satisfaction_rating
EMP001,Sales,Asia Pacific,85.88,3,3.6
EMP002,IT,Asia Pacific,65.99,1,4.7
EMP003,R&D,Europe,64.03,1,3.1
EMP004,Marketing,Asia Pacific,80.4,3,3.9
EMP005,Marketing,Latin America,81.85,15,3.4
"""

df = pd.read_csv(io.StringIO(csv_data))

it_count = (df["department"] == "IT").sum()
print("IT department frequency:", it_count)

plt.figure(figsize=(8, 6))
ax = sns.countplot(data=df, x="department", order=df["department"].value_counts().index, palette="Set2")
ax.set_title("Department Distribution")
ax.set_xlabel("Department")
ax.set_ylabel("Count")
plt.tight_layout()
plt.savefig("chart.png", dpi=144, bbox_inches="tight")
